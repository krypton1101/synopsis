enum color {red,green,blue};
class monstr
{
	int health, ammo;
	color skin;
	char* name;
public:
	monstr(int health=100, int ammo=10);
	monstr(color skin);
	monstr(char* name);
	monstr(const monstr &M); // copy
	int get_health() {return health;}
	int get_ammo() {return ammo;}
};

monstr::monstr(int health=100, int ammo=10)
{
	this->health = health;
	this->ammo = ammo;
}
monstr::monstr(color skin)
{
	switch(skin)
	{
	case red: health=100; ammo=100; name=0; break;
	case green: health=50; ammo=50; name=0; break;
	case blue: health=10; ammo=10; name=0; break;
	}
}
monstr::monstr(char* name)
{
	name = new char[strlen(name)+1];
	strcpy(this->name, name);
	health=100; ammo=100; skin=red;
}
monstr::monstr(const monstr &M)
{
	if(M.name)
	{
		name = new char[strlen(M.name)+1];
		strcpy(name, M.name);
	}
	else name=0;
	health=M.health;
	ammo=M.ammo;
	skin=M.skin;
}

int main()
{
	monstr Green(green);
	monstr *m = new monstr("Ork");

	monstr Vasya(blue);
	monstr Super=Vasya;
	monstr *m = new monstr("Ork");
	monstr Green = *m;
}

#include <iostream>
using namespace std;

#define SIZE 10

class stack
{
	char stck[SIZE];
	int top;
public:
	void init();
	void push(char ch);
	char pop();
};

void stack::init()
{
	top=0;
}
void stack::push(char ch)
{
	if(top==SIZE)
	{
		cout << "Stack is full" << endl;
		return;
	}
	stck[top] = ch;
	top++;
}
char stack::pop()
{
	if(!top)
	{
		cout << "Stack is empty" << endl;
		return 0;
	}
	//
	top--;
	return stck[top+1];
}

int main()
{
	stack s1,s2;
	int i;
	s1.init();
	s2.init();
	s1.push('a');
	s2.push('x');
	s1.push('g');
	s2.push('v');
	s1.push('h');
	s2.push('l');

	cout << s1.pop() << " " << s2.pop << endl;
}
