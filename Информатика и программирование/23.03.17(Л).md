# Динамические структуры
```c++
#include <iostream>
using namespace std;
struct Node
{
	Node *prev, *next;
	int d;
};
Node *first(int d);
void add(Node **pend, int d);
Node *find(Node *const pbeg, int i);
bool remove(Node **pbeg, Node **pend, int key);
Node *insert(Node *const pbeg, Node **pend, int key, int d);
void add_sort(Node **pbeg, Node **pend, int d);

int main()
{
	Node *pbeg = first(1);
	Node *pend = pbeg;
	for(int i=2; i<6; i++)
	{
		add(&pbeg, i);
	}
	insert(pbeg, &pend, 2, 200);
	if(!remove(&pbeg, &pend, 5))
	{
		cout << "nnot found" << endl;
	}
	Node *pv = pbeg;
	while(pv)
	{
		cout << pv->d << ' ' << endl;
		pv = pv->next;
	}
	return 0;
}

Node *first(int d)
{
	Node *pv = new Node;
	pv->d = d;
	pv->next = 0;
	pv->prev = 0;
	return pv;
}
void add(Node **pend, int d)
{
	Node *pv = new Node;
	pv->d = d;
	pv->next = 0;
	pv->prev = *pend;
	(*pend)->next = pv;
	*pend = pv;
}
Node *find(Node *const pbeg, int i)
{
	Node *pv = pbeg;
	while(pv)
	{
		if(pv->d == i) break;
		pv = pv->next;
	}
	return pv;
}
bool remove(Node **pbeg, Node **pend, int key)
{
	if(Node *pkey = find(*pbeg, key))
	{
		if(pkey==*pbeg)
		{
			*pbeg = (*pbeg)->next;
			(*pbeg)->prev = 0;
		}
		else if(pkey == *pend)
		{
			*pend = (*pend)->prev;
			(*pend)->next = 0;
		}
		else
		{
			(pkey->prev)->next = pkey->next;
			(pkey->next)->prev = pkey->prev;
		}
		delete pkey;
		return true;
	}
	return false;
}
Node *insert(Node *const pbeg, Node **pend, int key, int d)
{
	Node *pv = new Node;
	pv->d = d;
	Node *pkey = find(*pbeg, key);
	pv->next = pkey->next;
	pv->prev = pkey;
	pkey->next = pkey;
	if(pkey == *pend) *pend = pv;
	else (pkey->next)->prev = pkey;
	return pv;
}
void add_sort(Node **pbeg, Node **pend, int d)
{
	Node *pv = new Node;
	pv->d = d;
	Node *pt = *pbeg;
	while(pt)
	{
		if(d<pt->d)
		{
			pv->next = pt;
			if(pt==*pbeg)
			{
				pt->prev = 0;
				*pbeg = pv;
			}
			else //in mid
			{
				(pt->prev)->next = pv;
				pv->prev = pt->prev;
			}
			pt->prev = pv;
			return;
		} //if(d<pt->d)
		pt = pt->next;
	}
	pv->next = 0;
	pv->prev = *pend;
	(*pend)->next = pv;
	*pend = pv;
}
```
