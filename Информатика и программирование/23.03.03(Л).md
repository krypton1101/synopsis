```c++
			cin >> a[i][j];
		}
	}

		for(j=0;j<ncol;j++){
			cout << setw(4) << a[i][j];
		}
	}

	for(i=0;i<ncol;i++){
		flag=true;
		for(j=0;j<nrow;j++){
			if(flag) flag = a[j][i] > 0;
			else break;
		}
		if(flag){
			cout << "Index of left positive col: " << i << endl;
			return 0;
		} 
	}
	cout << "No positive rows" << endl;
	return 0;
}
```

```c++
void swap(int x, int y){
	int temp = x;
	x = y;
	y = temp;
}
int i = 1, j = 2;
swap(i, j);
cout << i << j; //1 2

void swap(int *x, int *y){
	int temp = *x;
	*x = *y;
	*y = temp;
}
int i = 1, j = 2;
swap(&i, &j);
cout << i << j; //2 1

void swap(int &x, int &y){
	int temp = x;
	x = y;
	y = temp;
}
int i = 1, j = 2;
swap(i, j);
cout << i << j; //2 1
```

```c++
int sum(const int *a, const int nstr, const int nstlb){
	int i,j,s=0;
	for(i=0; i<nstr; i++){
		for(j=0; j<nstlb; j++){
			s+=a[i*nstlb+j];
		}
	}
	return s;
}
```
