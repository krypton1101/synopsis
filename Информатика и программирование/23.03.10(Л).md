{
	int sum = 0, i, j;

	{

	}

```c++
#include <iostream>
using namespace std;
template <class Type> void sort(Type *b, int n);
int main()
{
	const int n = 20;
	int i, b[n];
	for(i=0; i<n; i++)
	{
		cin >> b[i];
	}
	sort(b, n);
	for(i=0;i<n;i++) cout << b[i] <<" ";
	double a[] = {0.22, 112, 321.54, 43.65, 65, 0.254};
	sort(a, 6);
	for(i=0;i<6;i++) cout << a[i] <<" ";
	return 0;
}

template <class Type>
void sort(Type *b, int n)
{
	Type a;
	for(int i=0; i<n; i++)
	{
		int imin = i;
		for(int j=i+1; j<n; j++)
		{
			if(b[j]<b[imin]) imin = j;
		}
		a=b[i];
		b[i]=b[imin];
		b[imin]=a;
	}
}
```
